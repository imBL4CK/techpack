StartupEvents.registry("item", (e) => {
  let natural = [
    "iron",
    "copper",
    "gold",
    "diamond",

    "tin",
    "silver",
    "nickel",
    "lead",
    "platinum",
    "aluminum",
    "chrome",
  ];
  natural.forEach((registry) => {
    e.create(`techpack:native_${registry}_cluster`)
      .texture(`techpack:item/library/native_${registry}_cluster`)
      .tag("techpack:native_clusters")
      .tag(`techpack:native_clusters/${registry}`);
  });
});
